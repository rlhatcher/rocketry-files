group per immediate subfolder listing that subtree's files (recursively). Param: dir (defaults to the current page's directory). {%- endcomment -%} {%- assign dir = include.dir | default: page.dir -%} {%- assign dlen = dir | size -%} {%- assign repo = "https://github.com/rlhatcher/rocketry-files" -%} {%- assign files = site.static_files | sort: "path" -%} {%- comment -%} immediate child directories {%- endcomment -%} {%- assign diracc = "" -%} {%- for f in files -%} {%- assign prefix = f.path | slice: 0, dlen -%} {%- if prefix == dir -%} {%- assign rel = f.path | remove_first: dir -%} {%- if rel contains "/" -%} {%- assign seg = rel | split: "/" | first -%} {%- unless seg == "assets" -%}{%- assign diracc = diracc | append: seg | append: "," -%}{%- endunless -%} {%- endif -%} {%- endif -%} {%- endfor -%} {%- assign dirs = diracc | split: "," | uniq | sort -%} {%- if dir == "/" -%} {%- for d in dirs -%} {%- assign purl = "/" | append: d | append: "/" -%} {%- assign ppage = site.pages | where: "url", purl | first -%} {%- if ppage -%}{%- assign dhref = purl | relative_url -%}{%- else -%}{%- assign dpath = "/" | append: d | replace: " ", "%20" -%}{%- assign dhref = repo | append: "/tree/main" | append: dpath -%}{%- endif -%} {%- endfor -%} {%- endif -%} {%- for f in files -%} {%- assign prefix = f.path | slice: 0, dlen -%} {%- if prefix == dir -%} {%- assign rel = f.path | remove_first: dir -%} {%- unless rel contains "/" -%}{%- unless f.extname == ".md" -%}{%- include repo-row.html file=f label=rel -%}{%- endunless -%}{%- endunless -%} {%- endif -%} {%- endfor -%}
